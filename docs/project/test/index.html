function addSerial1(arr) {
  let result = [];
  // do something
  const len = arr.length;
  const obj = {};
  for (let i = 0; i < len; i++) {
    if (obj[arr[i]]) {
      obj[arr[i]].number += 1;
      arr[i] = arr[i] + obj[arr[i]].number;
    } else {
      obj[arr[i]] = {};
      obj[arr[i]].number = 1;
      obj[arr[i]].index = i;
    }
  }
  for (const key in obj) {
    if (obj[key].number > 1) {
      arr[obj[key].index] = `${arr[obj[key].index] }1`;
    }
  }
  result = arr;
  return result;
}

    if (arr[i][j] === -1) {
